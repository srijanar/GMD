<?php
function getDatabaseConnection(){
    $host = "localhost";
    $userName = "root";
    $password = "root";
    $dbName = "gmd";
    $mysqli = new mysqli($host, $userName, $password, $dbName);
    if(!$mysqli){
        return mysqli_connect_error();
    }
    else{
        return $mysqli;
    }
}

function closeDatabaseConnection($connection){
    $connection->close();
}
